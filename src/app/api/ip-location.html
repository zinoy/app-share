API 说明

根据用户IP获取省与市，只限于中国省与市，不包括国外ip。

参数说明:
  • format : 当前可用的值为js，默认是直接返回JSON格式。

回调返回参数说明:
  • 默认返回 : {"ip":"xxx.xxx.xxx.xxx","province":"上海","city":"上海"}
  • format=js : 返回如：var hylinkProvinceCity = {"ip":"xxx.xxx.xxx.xxx","province":"上海","city":"上海"};

方式A：

<script src="https://www.chevrolet.com.cn/apinew/ip_location.aspx?format=js"></script> <!--返回var hylinkProvinceCity = {"ip":"xxx.xxx.xxx.xxx","province":"上海","city":"上海"};，可直接使用hylinkProvinceCity变量-->

方式B：

//通过ajax方式去获取时间。 $.getJSON("https://www.chevrolet.com.cn/apinew/ip_location.aspx").done(function(data){ console.log(data); }).fail(function(){console.error("ip_location.aspx Error!")});

实现代码：

/***************************************根据IP获取城市地址**********************************************************/ getUserProvCity = function(cbFn){ //cbFn('广西','柳州');return; var prov,city; readProvAndCitySession(); if(prov!=null && city!=null){ cbFn(prov, city); return;} //如果页面中有引用百度地图api，则会通过百度地图api获取省市https://api.map.baidu.com/api?v=2.0&ak=your-api-key&s=1 if(window['BMap'] && window['BMap']['LocalCity'] && window['BMap']['Geocoder']){//百度地图已加载 new BMap.LocalCity().get(function(result){ new BMap.Geocoder().getLocation(result.center, function(result){ prov = result.addressComponents.province; city = result.addressComponents.city; saveProvAndCitySession(prov, city); }); }); }else{//页面中无百度地图API引用，调用服务器上的接口 $.getScript('https://www.chevrolet.com.cn/apinew/ip_location.aspx?format=js').done(function(data, status, jqxhr){ if(status=='success' && window.hylinkProvinceCity){ prov = window.hylinkProvinceCity.province; city = window.hylinkProvinceCity.city; if(prov!="" && city!="") saveProvAndCitySession(prov, city); window.hylinkProvinceCity = null; }else{// //alternateGetProvCity(cbFn); } }).fail(function(){ //alternateGetProvCity(cbFn); }); } //把省与市保存到到sessionStorage中，下次再调用时，直接从sessionStorage中获取 function saveProvAndCitySession(prov, city){ cbFn(prov, city); if(window.sessionStorage){ try{ sessionStorage.setItem('userLocateProvince', prov); sessionStorage.setItem('userLocateCity', city); }catch(err){}; } } //从seesionStorage中获取省市 function readProvAndCitySession(){ if(window.sessionStorage){ try{ prov = sessionStorage.getItem('userLocateProvince'); city = sessionStorage.getItem('userLocateCity'); }catch(err){}; } } }
