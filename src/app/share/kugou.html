if(window.KgMobileCall && KgMobileCall.isInClient()){ KgMobileCall.share({ "shareName" : _shareDef.title, "hash" : "", "listID" : "", "type" : 3, "shareData" : { "linkUrl" : encodeURIComponent(_shareDef.url), "picUrl" : _shareDef.img, "content": _shareDef.desc, "title" : _shareDef.title }, "suid" : "", "slid" : "", "imgUrl" : _shareDef.img, "filename" : "", "duration": 0 }); }

酷狗音乐APP底部有音乐条，通过以下代码可以隐藏音乐条，并暂停播放音乐。

if(window.KgMobileCall && KgMobileCall.isInClient()) { KgMobileCall.showHideBar({state: 0}); KgMobileCall.playOrPause({type: 3}, function(){ KgMobileCall.showHideBar({state: 0}); }); }
